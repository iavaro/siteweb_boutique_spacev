How to Use API Key

Add the following code to your FiveM server's integration script:

-- SpaceV API Configuration
Config.SpaceV = {
  apiKey = "YOUR_API_KEY_HERE",
  serverUrl = "YOUR_SERVER_URL"
}

-- Send player data to SpaceV
RegisterNetEvent('spacev:updatePlayer')
AddEventHandler('spacev:updatePlayer', function(playerData)
  PerformHttpRequest(Config.SpaceV.serverUrl .. '/api/fivem/player/update-stats', function(err, response)
    -- Handle response
  end, 'POST', json.encode({
    apiKey = Config.SpaceV.apiKey,
    license = playerData.license,
    playerName = playerData.name,
    stats = playerData.stats
  }), {['Content-Type'] = 'application/json'})
end)
